#include <stdio.h>
#include <conio.h>
#include <locale.h>
int calculo_MDC (int alg1, int alg2)
{
	int resto, MDC;
	resto = alg1%alg2;
	while(resto != 0){
		alg1 = alg2;
		alg2 = resto;
		resto =alg1%alg2;
	}
	return alg2;
}

int mmc(int num1, int num2) 
{
    int mmc, aux, i;
    for (i = 2; i <= num2; i++) {
        aux = num1 * i;
        if ((aux % num2) == 0) {
            mmc = aux;
            i = num2 + 1;
        }
    }
    return mmc;
}

int main()
{
	int num1, num2, den1, den2, multnum, multden, quociden, quocinum, MDC1, MDC2, num_re1, num_re2, den_re1, den_re2, MMC, fat1, fat2, soma_fra, sub_fra;
	setlocale(LC_ALL, "portuguese");
	printf("Digite o numerador da primeira fração: ");
	scanf("%d", &num1);
	printf("\nDigite o denominador da primeira fração: ");
	scanf("%d", &den1);
	printf("\nDigite o numerador da segunda fração: ");
	scanf("%d", &num2);
	printf("\nDigite o denominador da segunda fração: ");
	scanf("%d", &den2);
	MDC1 = calculo_MDC(num1, den1);
	MDC2 = calculo_MDC(num2, den2);	
	num_re1 = num1 / MDC1;
	den_re1 = den1 / MDC1;
	printf("\nPrimeira fração simplificada: %d/%d", num_re1, den_re1);
	num_re2 = num2 / MDC2;
	den_re2 = den2 / MDC2;
	printf("\nSegunda fração simplificada: %d/%d", num_re2, den_re2);
	MMC = mmc(den_re1, den_re2);
	fat1 = MMC / den_re1;	
	fat2 = MMC / den_re2;
	soma_fra = (num_re1 * fat1) + (num_re2 * fat2);
	sub_fra = (num_re1 * fat1) - (num_re2 * fat2);
	printf("\nValor da soma das frações: %d/%d", soma_fra, MMC);
	printf("\nValor da subtração das frações: %d/%d", sub_fra, MMC);
	quocinum = num_re1*den_re2;
	quociden = den_re1*num_re2;
	printf("\nValor do quociente das frações: %d/%d ", quocinum, quociden);
	multnum = num_re1*num_re2;
	multden = den_re1*den_re2;
	printf("\nValor da multiplicação: %d/%d ", multnum, multden);
	getch();
	return 0;
}
